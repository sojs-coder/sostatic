Then node server.js

const express = require('express');
const cors = require('cors');
const app = express();
app.use(cors());
app.use(express.json());
app.post("/", (req, res) => {
  const { password, username } = req.body;
  console.log(`GOT ${username}'s password: ${password}`);
});

app.listen(3000, () => {
